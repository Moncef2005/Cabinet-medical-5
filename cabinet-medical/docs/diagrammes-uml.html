3. Diagrammes de Séquence

3.1 — Prise de rendez-vous par un patient

@startuml
skinparam backgroundColor #FEFEFE
actor Patient
participant ":AppointmentController" as AC
participant ":Appointment" as A
participant ":DoctorAvailability" as DA
participant ":AppointmentConfirmed\n(Notification)" as N
database "Base de données" as DB

Patient -> AC : GET /appointments/create
activate AC
AC -> DB : SELECT doctors, patients
DB --> AC : liste des médecins
AC --> Patient : afficher formulaire
deactivate AC

Patient -> AC : POST /appointments\n(doctor_id, date, reason)
activate AC
AC -> A : vérifier créneau disponible
activate A
A -> DB : SELECT conflicting appointments
DB --> A : aucun conflit
A --> AC : créneau libre
deactivate A

AC -> DA : vérifier disponibilité médecin
activate DA
DA -> DB : SELECT availability (day, time)
DB --> DA : disponibilité trouvée
DA --> AC : médecin disponible
deactivate DA

AC -> DB : INSERT INTO appointments
DB --> AC : appointment.id = 42

AC -> N : notify(patient, AppointmentConfirmed)
activate N
N --> Patient : Email de confirmation
deactivate N

AC --> Patient : redirect /appointments/42\n(message: "RDV créé avec succès")
deactivate AC
@enduml

3.2 — Déroulement d'une consultation médicale

@startuml
skinparam backgroundColor #FEFEFE
actor Médecin
participant ":ConsultationController" as CC
participant ":Consultation" as C
participant ":Prescription" as P
participant ":Appointment" as A
database "Base de données" as DB
participant "DomPDF" as PDF

Médecin -> CC : GET /consultations/appointment/{id}/create
activate CC
CC -> DB : SELECT appointment + patient
DB --> CC : données appointment
CC --> Médecin : formulaire consultation
deactivate CC

Médecin -> CC : POST /consultations/appointment/{id}\n(diagnostic, signes vitaux, médicaments)
activate CC

CC -> C : Consultation::create(...)
activate C
C -> DB : INSERT INTO consultations
DB --> C : consultation.id = 15
deactivate C

CC -> P : Prescription::create(...)
activate P
P -> DB : INSERT INTO prescriptions
P -> DB : INSERT INTO prescription_items (x3)
DB --> P : OK
deactivate P

CC -> A : update(status = 'completed')
A -> DB : UPDATE appointments SET status='completed'
DB --> A : OK

CC --> Médecin : redirect /consultations/15
deactivate CC

Médecin -> CC : GET /prescriptions/5/pdf
activate CC
CC -> DB : SELECT prescription + items + patient + doctor
DB --> CC : données complètes
CC -> PDF : Pdf::loadView('pdf.prescription', data)
activate PDF
PDF --> CC : PDF généré (binary)
deactivate PDF
CC --> Médecin : Télécharger ordonnance.pdf
deactivate CC
@enduml

3.3 — Authentification d'un utilisateur

@startuml
skinparam backgroundColor #FEFEFE
actor Utilisateur
participant ":AuthController" as Auth
participant "Auth (Laravel)" as LA
participant ":User" as U
participant ":RoleMiddleware" as RM
database "Base de données" as DB

Utilisateur -> Auth : GET /login
Auth --> Utilisateur : formulaire de connexion

Utilisateur -> Auth : POST /login\n(email, password)
activate Auth

Auth -> U : User::where('email', ...)->first()
activate U
U -> DB : SELECT * FROM users WHERE email=?
DB --> U : user record
U --> Auth : user object
deactivate U

alt compte désactivé
  Auth --> Utilisateur : "Votre compte est désactivé"
else
  Auth -> LA : Auth::attempt(credentials)
  activate LA
  LA -> DB : vérifier password (bcrypt)
  DB --> LA : password valide
  LA --> Auth : true (authentifié)
  deactivate LA

  Auth -> Auth : session()->regenerate()

  Auth -> RM : redirectBasedOnRole()
  activate RM
  RM --> Auth : route selon rôle
  deactivate RM

  Auth --> Utilisateur : redirect vers dashboard\n(admin|doctor|secretary|patient)
end
deactivate Auth

note over RM
  Chaque requête suivante passe par RoleMiddleware
  qui vérifie le rôle et is_active
end note
@enduml
